on_game_start_after_lobby= {
    on_actions = {
        delay = { months = 1 }
        T4N_on_monthly_tribute
    }
}

T4N_on_monthly_tribute = {
    effect = {
        every_suzerain = {
            # Determine if still a suzerain
            trigger_event = { id = tributaries.4 }
            # Income
            trigger_event = { id = tributaries.5 }
        }
        trigger_event = {
            on_action = T4N_on_monthly_tribute
            months = 1
        }
    }
}


on_war_won_attacker = {
	# The war notifications are handled in `effect` instead of `events` due to order of operations.
	# `effect` fires on THIS tick, `events` fires on the NEXT tick, and the war gets destroyed between this tick and the next.
	on_actions = {
		T4N_tributaries_on_war_won_attacker
	}
}

T4N_tributaries_on_war_won_attacker = {
	effect = {
        if = {
            limit = {
                scope:defender = {
                    OR = {
                        is_tributary = yes
                        is_suzerain = yes
                    }
                }
            }
            scope:attacker = {
                save_scope_as = tributaries_slash_suzerain_war_winner
            }
            scope:defender = {
                save_scope_as = tributary_slash_suzerain_loser
            }
            scope:tributaries_slash_suzerain_war_winner = {
                trigger_event = { id = tributaries.3 days = 1 }
            }
        }
	} 
}


# A title is lost by a character
# root = the old holder
# scope:title = the title that changes hands
# scope:new_holder = new holder.
on_title_lost = {
    on_actions = {
        T4N_tributaries_on_title_lost
    }
}

T4N_tributaries_on_title_lost = {
    effect = {
        save_scope_as = temp_scope
        switch = {
            trigger = exists

            var:my_suzerain = {
                if = {
                    limit = {
                        OR = {
                            is_landed = no
                            is_independent_ruler = no
                            liege = scope:new_holder
                        }
                    }
                    if = {
                        limit = {
                            is_permanent_tributary = {
                                SUZERAIN = scope:temp_scope.var:my_suzerain
                                TRIBUTARY = scope:temp_scope
                            }
                        }
                        unset_tributary = {
                            TRIBUTARY = scope:temp_scope
                        }                
                    }
                    else = {
                        unset_tributary = {
                            TRIBUTARY = scope:temp_scope
                        }
                    }
                }
            }
            var:suzerain = {
                if = {
                    limit = {
                        OR = {
                            is_landed = no
                            is_independent_ruler = no
                            liege = scope:new_holder
                        }
                    }
                    every_in_list = {
                        variable = permanent_tributaries
                        set_variable = { 
                            name = tributary_succ_var
                            value = this
                        }
                        pass_tributary_type_to_heir = {
                            SCOPE = var:tributary_succ_var
                            TRIBUTARY = var:tributary_succ_var
                            SUZERAIN = scope:new_holder
                        }
                        remove_variable = tributary_succ_var
                    }
                    every_in_list = {                    
                        variable = non_permanent_tributaries
                        set_variable = { 
                            name = tributary_non_perma_succ_var
                            value = this
                        }                        
                        unset_tributary = { TRIBUTARY = var:tributary_non_perma_succ_var }
                        remove_variable = tributary_non_perma_succ_var
                    }
                    clear_variable_list = non_permanent_tributaries
                }
            }
        }
    }
}

# character just about to die in root scope
# if a killer is know, it's set as scope:killer
#Triggered by code
on_death = {
    on_actions = {
        T4N_tributaries_on_death
    }
}

T4N_tributaries_on_death = {
    effect = {
        if = { 
            limit = {
                has_realm_law_flag = partition_succession_law 
                root = {
                    NOT = { exists = var:my_suzerain }
                }
            }
            root.primary_heir = {
                save_scope_as = main_heir
            }
            ordered_heir = {       
                order_by = age
                limit = {
                    NOT = { this = root.primary_heir }
                    any_heir_to_title = {
                        tier == root.primary_title.tier
                    }
                }
                save_scope_as = heir
                make_tributary = {
                    TYPE = tributary_karling
                    TRIBUTARY = scope:heir
                    SUZERAIN = scope:main_heir
                }                              
            }
            debug_log = "Partition Succesion happened in which newly formed realms became tributaries of primary heir"
        }      
        if = {
            limit = {
                root = {
                    exists = var:suzerain
                }
            }
            save_scope_as = suzerain
			#Copy old Ruler tributaries list to his primary heir (on_death)
			#Root is old ruler
            every_in_list = {
				variable = permanent_tributaries
                set_variable = { 
                   name = tributary_succ_var
                   value = this
                }
                pass_tributary_type_to_heir = {
                    SCOPE = var:tributary_succ_var
                    TRIBUTARY = var:tributary_succ_var
                    SUZERAIN = scope:suzerain.primary_heir
                }
                remove_variable = tributary_succ_var
			}		
			every_in_list = {
				variable = non_permanent_tributaries
                set_variable = { 
                    name = tributary_succ_var_np
                    value = this
                }
                unset_tributary = {
                    TRIBUTARY = var:tributary_succ_var_np
                }
                remove_variable = tributary_succ_var_np
			}	          
			remove_variable = suzerain       
        }
        if = {
            limit = {
                root = {
                    exists = var:my_suzerain
                }
            }
            save_scope_as = temp_scope
            var:my_suzerain = {
                save_scope_as = suzerain
            }
            root.primary_heir = {
                save_scope_as = tributary
            }
            debug_log = "I passed my_suzerain"
			#give the heir var:my_suzerain
			#Root is old tributary
            if = {
                limit = {
                    is_permanent_tributary = {
                        SUZERAIN = scope:suzerain
                        TRIBUTARY = scope:temp_scope
                    }
                }
                debug_log = "I passed trib type check"
                pass_tributary_type_to_heir = {
                    SCOPE = scope:temp_scope
                    TRIBUTARY = scope:tributary
                    SUZERAIN = scope:suzerain
                }               
                unset_tributary = { TRIBUTARY = scope:temp_scope }
                if = { 
                    limit = {
                        has_realm_law_flag = partition_succession_law 
                    }
                    ordered_heir = {       
                        order_by = age
                        limit = {
                            NOT = { this = root.primary_heir }
                            any_heir_to_title = {
                                tier == root.primary_title.tier
                            }
                        }
                        set_variable = { 
                            name = succ_trib_var
                            value = this
                        }
                        save_scope_as = heir
                        make_tributary = {
                            TYPE = tributary_karling
                            TRIBUTARY = scope:heir
                            SUZERAIN = var:my_suzerain
                        }                                       
                    }
                    debug_log = "Partition Succesion happened in which newly formed realms became tributaries of character former suzerain"
                }                  
                debug_log = "I Passed tributary pass effect"
                unset_tributary = {
                    TRIBUTARY = root
                }
                debug_log = "I passed unset"
                remove_variable = tributary_succ_var              
            }
            else = {
                set_variable = { 
                    name = tributary_succ_var_np
                    value = this
                }
                unset_tributary = {
                    TRIBUTARY = var:tributary_succ_var_np
                }
                remove_variable = tributary_succ_var_np
            }
        }
    }
}
